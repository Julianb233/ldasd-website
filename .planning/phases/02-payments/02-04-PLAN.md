'use client'

import { useState } from 'react'
import { createCheckoutSession } from '@/lib/actions/checkout'
import type { ProductType, PricingTier } from '@/lib/stripe/products'

// Loading spinner component
function Spinner({ className }: { className?: string }) {
  return (
    <svg
      className={`animate-spin ${className}`}
      xmlns="http://www.w3.org/2000/svg"
      fill="none"
      viewBox="0 0 24 24"
    >
      <circle
        className="opacity-25"
        cx="12"
        cy="12"
        r="10"
        stroke="currentColor"
        strokeWidth="4"
      />
      <path
        className="opacity-75"
        fill="currentColor"
        d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"
      />
    </svg>
  )
}

export type CheckoutButtonProps = {
  productType: ProductType
  pricingTier: PricingTier
  children?: React.ReactNode
  className?: string
  variant?: 'primary' | 'secondary'
  size?: 'default' | 'large'
  disabled?: boolean
}

export function CheckoutButton({
  productType,
  pricingTier,
  children = 'Get Started',
  className = '',
  variant = 'primary',
  size = 'default',
  disabled = false,
}: CheckoutButtonProps) {
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  // Base styles
  const baseStyles = 'inline-flex items-center justify-center gap-2 rounded-full font-semibold transition-all duration-300 disabled:opacity-50 disabled:cursor-not-allowed'

  // Variant styles
  const variantStyles = {
    primary: 'bg-secondary text-white hover:bg-accent',
    secondary: 'bg-white text-foreground border-2 border-foreground/20 hover:border-foreground/40',
  }

  // Size styles
  const sizeStyles = {
    default: 'px-6 py-3 text-base',
    large: 'px-8 py-4 text-lg',
  }

  async function handleCheckout() {
    setLoading(true)
    setError(null)

    try {
      const result = await createCheckoutSession(productType, pricingTier)

      if (result.error) {
        setError(result.error)
        setLoading(false)
        return
      }

      if (result.url) {
        // Redirect to Stripe Checkout
        window.location.href = result.url
        // Keep loading state while redirecting
      }
    } catch (err) {
      console.error('Checkout error:', err)
      setError('An unexpected error occurred. Please try again.')
      setLoading(false)
    }
  }

  return (
    <div className="flex flex-col items-start">
      <button
        onClick={handleCheckout}
        disabled={loading || disabled}
        className={`${baseStyles} ${variantStyles[variant]} ${sizeStyles[size]} ${className}`}
      >
        {loading ? (
          <>
            <Spinner className="w-5 h-5" />
            <span>Processing...</span>
          </>
        ) : (
          children
        )}
      </button>
      {error && (
        <p className="mt-2 text-sm text-red-600" role="alert">
          {error}
        </p>
      )}
    </div>
  )
}
```

Key points:
- 'use client' for interactivity
- Calls server action directly
- Shows loading spinner during checkout
- Displays error message on failure
- Keeps loading while redirecting to Stripe
- Customizable via props (variant, size)
- Accessible error message with role="alert"

'use client'

import { CheckoutButton } from './CheckoutButton'
import {
  DISPLAY_PRICES,
  PRODUCT_NAMES,
  formatPrice,
  hasCouplesPricing,
  type ProductType,
  type PricingTier,
} from '@/lib/stripe/products'

// Check icon for features
function CheckIcon({ className }: { className?: string }) {
  return (
    <svg className={className} fill="none" viewBox="0 0 24 24" strokeWidth={2} stroke="currentColor">
      <path strokeLinecap="round" strokeLinejoin="round" d="M5 13l4 4L19 7" />
    </svg>
  )
}

export type ProductCardProps = {
  productType: ProductType
  pricingTier: PricingTier
  features: string[]
  description?: string
  popular?: boolean
  className?: string
}

export function ProductCard({
  productType,
  pricingTier,
  features,
  description,
  popular = false,
  className = '',
}: ProductCardProps) {
  const displayPrice = DISPLAY_PRICES[productType]?.[pricingTier]
  const productName = PRODUCT_NAMES[productType]
  const supportsCouples = hasCouplesPricing(productType)

  // If couples selected but product doesn't support it, show individual price
  const effectiveTier = supportsCouples ? pricingTier : 'individual'
  const effectivePrice = DISPLAY_PRICES[productType]?.[effectiveTier] || 0

  return (
    <div
      className={`relative bg-white rounded-3xl border ${
        popular ? 'border-secondary shadow-premium' : 'border-gray-200'
      } p-8 flex flex-col ${className}`}
    >
      {/* Popular Badge */}
      {popular && (
        <div className="absolute -top-4 left-1/2 -translate-x-1/2">
          <span className="bg-secondary text-white text-sm font-semibold px-4 py-1.5 rounded-full">
            Most Popular
          </span>
        </div>
      )}

      {/* Header */}
      <div className="mb-6">
        <h3 className="text-xl font-bold text-foreground">{productName}</h3>
        {description && (
          <p className="text-foreground/70 mt-2">{description}</p>
        )}
      </div>

      {/* Price */}
      <div className="mb-6">
        <div className="flex items-baseline gap-1">
          <span className="text-4xl font-bold text-foreground">
            {formatPrice(effectivePrice)}
          </span>
          <span className="text-foreground/50">one-time</span>
        </div>
        {pricingTier === 'couples' && supportsCouples && (
          <p className="text-sm text-green-600 mt-1">
            Includes documents for both partners
          </p>
        )}
        {pricingTier === 'couples' && !supportsCouples && (
          <p className="text-sm text-gray-500 mt-1">
            Individual pricing only
          </p>
        )}
      </div>

      {/* Features */}
      <ul className="space-y-3 mb-8 flex-grow">
        {features.map((feature, index) => (
          <li key={index} className="flex items-start gap-3">
            <CheckIcon className="w-5 h-5 text-green-600 shrink-0 mt-0.5" />
            <span className="text-foreground/80">{feature}</span>
          </li>
        ))}
      </ul>

      {/* CTA */}
      <CheckoutButton
        productType={productType}
        pricingTier={effectiveTier}
        variant={popular ? 'primary' : 'secondary'}
        size="large"
        className="w-full"
      >
        Get Started
      </CheckoutButton>
    </div>
  )
}

// Simplified card for pricing page
export function ProductCardCompact({
  productType,
  pricingTier,
  className = '',
}: {
  productType: ProductType
  pricingTier: PricingTier
  className?: string
}) {
  const displayPrice = DISPLAY_PRICES[productType]?.[pricingTier]
  const productName = PRODUCT_NAMES[productType]
  const supportsCouples = hasCouplesPricing(productType)
  const effectiveTier = supportsCouples ? pricingTier : 'individual'
  const effectivePrice = DISPLAY_PRICES[productType]?.[effectiveTier] || 0

  return (
    <div className={`bg-white rounded-2xl border border-gray-200 p-6 ${className}`}>
      <div className="flex items-center justify-between mb-4">
        <h4 className="font-semibold text-foreground">{productName}</h4>
        <span className="text-2xl font-bold text-foreground">
          {formatPrice(effectivePrice)}
        </span>
      </div>
      <CheckoutButton
        productType={productType}
        pricingTier={effectiveTier}
        variant="secondary"
        className="w-full"
      >
        Purchase
      </CheckoutButton>
    </div>
  )
}
```

Key points:
- Uses DISPLAY_PRICES from products.ts
- Handles couples pricing fallback for products without it
- Popular badge for highlighted products
- Features list with check icons
- Two variants: full and compact
- Integrates CheckoutButton
- Responsive and accessible
